# Simple usage

<template>
    <div>
        <w-data-table :headers="headers" :items="items"></w-data-table>
    </div>
</template>

# Item slot

<template>
    <div>
        <w-data-table :headers="headers" :items="items">
          <template v-slot:item.first_name="{ item }">
                {{ item.first_name.toUpperCase() }}
          </template>  
        </w-data-table>
    </div>
</template>
