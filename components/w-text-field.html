# w-text-field

Name value Type string | number Default undefiend
Name label Type string Default undefiend
Name defaultLabelClass Type object | string | object Default [ "block text-gray-700 text-sm font-bold" ]
Name defaultInputClass Type object | string | object Default [ "appearance-none focus:outline-none", "shadow rounded w-full leading-tight", "transition-border-color duration-200", "py-2 px-3", "border" ]
Name color Type string Default gray-300
Name textColor Type string Default gray-800
Name focus:color Type string Default blue-500
Name focus:textColor Type string Default gray-800
Name error:color Type string Default red-500
Name rules Type object Default []

# Simple usage

My name is: 
<template>
  <div>
      <w-text-field v-model="name" label='Name' placeholder='Type your name' />
      <div class="text-center my-4">
        My name is: <strong>{{name}}</strong>
      </div>
  </div>
</template>

# Validation

<template>
  <div>
      <w-text-field
        v-model='model'
        label='Min 5 characters max 10 characters'
        :rules='rules'
        placeholder="Type a text" />
  </div>
</template>
